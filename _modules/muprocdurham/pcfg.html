def cat_pretty(trees, off=False, *args, **kwargs):
    if off:
        return '\n'.join([tree.pretty(*args, **kwargs) for tree in trees])
    else:
        return '\n'.join(
            list('|'.join(x) for x in zip(*[tree.pretty(*args, **kwargs).split('\n') for tree in trees]))
        )
